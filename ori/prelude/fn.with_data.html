Expand description

Create a new WithState that replaces the data with the state.

§Example

struct Data {
    // ...
}

fn ui() -> impl View<Data> {
    with_data(
        || 0,
        |count| {
            on_click(
                button(text!("Clicked {} time(s)", count)),
                |cx, count| {
                    *count += 1;
                    cx.rebuild();
                },
            )
        }
    )
}
