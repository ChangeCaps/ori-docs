Create a new WithState using S::default().

ยงExample

struct Data {
    // ...
}

fn ui() -> impl View<Data> {
    with_state_default(
        |count: &mut i32, _data| {
            on_click(
                button(text!("Clicked {} time(s)", count)),
                |cx, (count, _data)| {
                    *count += 1;
                    cx.rebuild();
                },
            )
        }
    )
}
